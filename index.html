function update(dt) { if (!isPlaying || isPaused || isGameOver) return; const now = performance.now(); // ✅ FIX: Added now definition // Jetpack physics if (player.jetpack && spaceHeld) { player.vy -= 0.9; if (player.vy < -8) player.vy = -8; if (Math.random() < 0.3) createParticles(player.x + player.w/2, player.y + player.h, 1, '#ff4500', 3); } else { player.vy += 0.65 * timeScale; } player.y += player.vy * timeScale; let onGround = player.y + player.h >= floorY; let inPit = false; for (let o of obstacles) { if (o.type === 'pit' && player.x + player.w/2 > o.x && player.x + player.w/2 < o.x + o.w) { inPit = true; break; } } if (inPit) { if (player.y > H + 50) { audio.playCrash(); screenShake = 20; moonMood = -1; createParticles(player.x+player.w/2, H, 30, '#e74c3c', 8); gameOver(); } } else if (onGround) { player.y = floorY - player.h; player.vy = 0; player.grounded = true; } else { player.grounded = false; } if (player.grounded) animFrame += 0.15 * timeScale; if (player.grounded && !inPit) { comboTimer += dt * timeScale; if (comboTimer > 3000 && combo < 5) { combo++; comboTimer = 0; const comboEl = document.getElementById('combo'); const comboValueEl = document.getElementById('comboValue'); if (comboEl && comboValueEl) { comboValueEl.textContent = combo; comboEl.classList.add('show'); setTimeout(() => comboEl.classList.remove('show'), 2000); } } } else { comboTimer = 0; } spawnObjects(); if (player.magnet) coins.forEach(c => { const dx=(player.x+player.w/2)-(c.x+c.w/2); const dy=(player.y+player.h/2)-(c.y+c.h/2); const d=Math.sqrt(dx*dx+dy*dy); if(d<200) { c.x+=dx*0.1; c.y+=dy*0.1; } }); shootingStars.forEach((s, i) => { s.x -= 10 * timeScale; s.y += 5 * timeScale; s.alpha -= 0.02; if(s.alpha <= 0) shootingStars.splice(i, 1); }); for (let i = obstacles.length - 1; i >= 0; i--) { let o = obstacles[i]; // Move complex obstacles if (o.type === 'flying_spike') { o.y = o.baseY + Math.sin(Date.now() * 0.004 + o.phase) * 30; } else if (o.type === 'moving_block') { o.x = o.baseX - (gameSpeed * timeScale * (now - lastObstacleTime)/1000) + Math.sin(Date.now() * 0.003 + o.phase) * o.range; } else { o.x -= gameSpeed * timeScale; } if (o.x + o.w < 0) obstacles.splice(i, 1); if (checkCollision(player, o)) { if (o.type === 'pit') { // Falling handled above } else if (player.shield) { player.shield = false; audio.playCrash(); screenShake = 10; obstacles.splice(i, 1); document.querySelector('.powerup-shield')?.remove(); } else { audio.playCrash(); screenShake = 20; moonMood = -1; createParticles(player.x+player.w/2, player.y+player.h/2, 30, '#e74c3c', 8); gameOver(); } } } const time = Date.now() / 200; for (let i = coins.length - 1; i >= 0; i--) { let c = coins[i]; c.x -= gameSpeed * timeScale; c.y += Math.sin(time + c.bob) * 0.4; if (c.x + c.w < 0) coins.splice(i, 1); if (checkCollision(player, c)) { const val = player.doubleScore ? 2 * combo : 1 * combo; runCoins += val; totalCoins += val; localStorage.setItem('vk_runner_coins', totalCoins); const coinsEl = document.getElementById('coins'); const totalCoinsEl = document.getElementById('totalCoins'); if (coinsEl) coinsEl.textContent = runCoins; if (totalCoinsEl) totalCoinsEl.textContent = totalCoins; audio.playCoin(); createParticles(c.x+c.w/2, c.y+c.h/2, 15, '#f1c40f', 6); coins.splice(i, 1); updateMission('collect_coins', 1); comboTimer = 0; } } for (let i = powerups.length - 1; i >= 0; i--) { let p = powerups[i]; p.x -= gameSpeed * timeScale; if (p.x + p.w < 0) powerups.splice(i, 1); if (checkCollision(player, p)) { activatePowerup(p.type); powerups.splice(i, 1); } } for (let i = particles.length - 1; i >= 0; i--) { let p = particles[i]; p.x += p.vx * timeScale; p.y += p.vy * timeScale; p.vy += p.gravity * timeScale; p.life -= dt * 0.002 * timeScale; p.size *= 0.97; if (p.life <= 0) particles.splice(i, 1); } score += (player.doubleScore ? 0.2 : 0.1) * combo * timeScale; const scoreEl = document.getElementById('score'); if (scoreEl) scoreEl.textContent = Math.floor(score); updateMission('run_distance', 0.1 * timeScale); if (screenShake > 0) { screenShake *= 0.9; if (screenShake < 0.5) screenShake = 0; } }
